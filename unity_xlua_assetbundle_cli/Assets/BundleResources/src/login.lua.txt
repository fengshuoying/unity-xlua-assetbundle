
function start()
    btnLogin:GetComponent("Button").onClick:AddListener(function()
      	local act = txtAccount:GetComponent("InputField").text
        print("点击登陆按钮V3")
       	if act == "" then return print("登陆账号不能为空") end

       	g_App.sendLoginHttpMsg("http://192.168.6.210:8000", "windows", { act = act}, function(err)

       		if err ~= nil then return print("登陆失败 "..err) end

       		dump(g_PlayerData)
       		dump(g_BagData)
       	end)
    end)
end
